CREATE TABLE `banqueenligne`.`comptecourant` (

  `idCompteCourant` INT NOT NULL AUTO_INCREMENT,

  `soldeCompteCourant` DECIMAL NULL,
 
 `RIB` VARCHAR(45) NULL,
 
 `idClient` INT NOT NULL,
 
 PRIMARY KEY (`idCompteCourant`),

  INDEX `idClient_idx` (`idClient` ASC),

  CONSTRAINT `idClient1`
 
   FOREIGN KEY (`idClient`)
 
   REFERENCES `banqueenligne`.`client` (`idClient`)
  
  ON DELETE NO ACTION
   
 ON UPDATE NO ACTION);
