CREATE TABLE `banqueenligne`.`comptetitre` (
 
 `idCompteTitre` INT NOT NULL AUTO_INCREMENT,

  `soldeCompteTitre` DECIMAL NULL,

  `RIB` VARCHAR(45) NULL,
 
 `idClient` INT NOT NULL,

  PRIMARY KEY (`idCompteTitre`),
 
 INDEX `idClient_idx` (`idClient` ASC),
 
 CONSTRAINT `idClient`
  
  FOREIGN KEY (`idClient`)
 
   REFERENCES `banqueenligne`.`client` (`idClient`)
  
  ON DELETE NO ACTION
  
  ON UPDATE NO ACTION);
